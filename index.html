If you see a CORS error in the browser, you’ll need to call the API from your own server (proxy). See the Proxy (Node.js) snippet inside this file.

What if CORS blocks the request?

Browsers block cross‑origin requests if the remote server doesn’t allow them. Host a tiny proxy and call that instead.

// server.js (Node + Express minimal proxy)

// 1) npm init -y && npm i express node-fetch@3 // 2) node server.js // 3) In the frontend, set API_BASE = "http://your-server:3000" import express from 'express'; import fetch from 'node-fetch';

const app = express(); const PORT = process.env.PORT || 3000;

app.get('/likes', async (req, res) => { try { const { uid, region } = req.query; if(!uid || !region) return res.status(400).json({ error: 'uid and region required' });

const url = `https://likes.ffgarena.cloud/api/v2/likes?uid=${encodeURIComponent(uid)}&amount_of_likes=100&auth=trial-7d®ion=${encodeURIComponent(region)}`;
const r = await fetch(url);
const data = await r.json();
res.set('Access-Control-Allow-Origin', '*');
res.json(data);

} catch(err){ res.status(500).json({ error: 'proxy_failed', message: String(err) }); } });

app.listen(PORT, () => console.log('Proxy running on', PORT));
